37/**
38 * Apply common attributes (id and class) to an element.
39 * @template T of Element
40 * @param T $element
41 * @return T
42 */
43function _applyCommon(Element $element, ?string $id, ?string $class): Element
44{
45    if ($id !== null) {

47    }
